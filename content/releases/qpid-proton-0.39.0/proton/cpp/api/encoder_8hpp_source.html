135 template <class T> encoder& operator<<(const map_cref<T>& x) {
136 internal::state_guard sg(*this);
137 *this << start::map();
138 for (typename T::const_iterator i = x.ref.begin(); i != x.ref.end(); ++i)
139 *this << i->first << i->second;
140 *this << finish();
141 return *this;
142 }

126 // The known/unknown integer type magic is required because the C++ standard is
127 // vague a about the equivalence of integral types for overloading. E.g. char is
128 // sometimes equivalent to signed char, sometimes unsigned char, sometimes
129 // neither. int8_t or uint8_t may or may not be equivalent to a char type.
130 // int64_t may or may not be equivalent to long long etc. C++ compilers are also
131 // allowed to add their own non-standard integer types like __int64, which may
132 // or may not be equivalent to any of the standard integer types.
133 //
134 // The solution is to use a fixed, standard set of integer types that are
135 // guaranteed to be distinct for overloading (see type_id_of) and to use template
136 // specialization to convert other integer types to a known integer type with the
137 // same sizeof and is_signed.
138 
139 // Map arbitrary integral types to known integral types.
140 template<size_t SIZE, bool IS_SIGNED> struct integer_type;
141 template<> struct integer_type<1, true> { typedef int8_t type; };
142 template<> struct integer_type<2, true> { typedef int16_t type; };
143 template<> struct integer_type<4, true> { typedef int32_t type; };
144 template<> struct integer_type<8, true> { typedef int64_t type; };
145 template<> struct integer_type<1, false> { typedef uint8_t type; };
146 template<> struct integer_type<2, false> { typedef uint16_t type; };
147 template<> struct integer_type<4, false> { typedef uint32_t type; };
148 template<> struct integer_type<8, false> { typedef uint64_t type; };

167 template <class From, class To> struct is_convertible : public sfinae {
168  static yes test(const To&);
169  static no test(...);
170  static const From& from;
171  // Windows compilers warn about data-loss caused by legal conversions. We
172  // can't use static_cast because that will cause a hard error instead of

174 #ifdef _WIN32
175 #pragma warning( push )
176 #pragma warning( disable : 4244 )

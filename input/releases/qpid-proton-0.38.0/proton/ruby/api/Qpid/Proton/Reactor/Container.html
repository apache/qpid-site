def create_receiver(context, opts=nil)
  c = context if context.is_a? Qpid::Proton::Connection
  unless c
    url = Qpid::Proton::uri context
    c = super_connect(url, opts)
    opts ||= {}
    opts[:source] ||= url.amqp_address
  end
  c.open_receiver opts
end

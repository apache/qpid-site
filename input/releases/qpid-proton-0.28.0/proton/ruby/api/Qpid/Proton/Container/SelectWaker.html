def reset
  @lock.synchronize do
    return unless @set
    @rd.read_nonblock(1) rescue nil
    @set = false
  end
end

def wake
  @lock.synchronize do
    return if @set        # Don't write if already has data
    @set = true
    @wr.write_nonblock('x') rescue nil
  end
end
